var rows = document.querySelectorAll('.mtq_clickable');
 rows.forEach(function(row) {
	 var buttonElement = row.querySelector('.option_index_number');
	 var cmarkerElement = row.querySelector('.mtq_correct_marker');
	 var wmarkerElement = row.querySelector('.mtq_wrong_marker');
	
	 row.addEventListener("click",function(){
		const iscorrect = row.getAttribute('data-value');

		 buttonElement.style.display = "none";
		if(iscorrect==1){
			cmarkerElement.style.display = "block";
			//markerElement.classList.add("mtq_correct_marker");
		}
		else{
			wmarkerElement.style.display = "block";
			//markerElement.classList.add("mtq_wrong_marker");
			showCorrectAnswer(row);
			}
	 });
		 
 });
 
 function showCorrectAnswer(selectedRow) {
	 const correctRow=findCorrectRow(selectedRow);
	 var correctButtonElement = correctRow.querySelector('.option_index_number');
	 var correctMarkerElement = correctRow.querySelector('.mtq_correct_marker');
	 correctButtonElement.style.display = "none"; // Hide the correct button
	 correctMarkerElement.style.display = "block"; // Show the correct marker
 }
 function findCorrectRow(selectedRow) {
        // Simple example: just find the row with the 'correct_option' class within the same question
        return selectedRow.closest('.answer_table').querySelector('.mtq_correct_marker').closest('.mtq_clickable');
      }
